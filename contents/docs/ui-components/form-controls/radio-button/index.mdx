    "use client";

    import React from "react";

    type RadioButtonProps = {
      id: string;
      name: string;
      label: string;
      value: string;
      defaultChecked?: boolean;
      disabled?: boolean;
      onChange?: (value: string) => void;

      label,
      value,
      defaultChecked = false,
      disabled = false,
      onChange,

      return (

          htmlFor={id}

            disabled ? "opacity-50 cursor-not-allowed" : ""
          }`}
        >

            type="radio"
            id={id}
            name={name}
            value={value}
            defaultChecked={defaultChecked}
            disabled={disabled}
            onChange={() => onChange?.(value)}
            className="h-5 w-5 accent-blue-600 focus:ring-blue-500"

## Setup & Usage

<Step>
  <StepItem title="Create a new Component">
    Create a new component in the `@/components/ui-library` directory with the file name `RadioButton.tsx`.
  </StepItem>

  <StepItem title="Copy the Full Code inside the component">
    Copy the full code from the "Full Code" tab above and paste it into the `RadioButton.tsx` file.
  </StepItem>

  <StepItem title="Use the Component">
    The RadioButton component can be used in two ways:

    1. Basic usage:
    ```tsx
    <RadioButton
      id="option1"
      name="group1"
      label="Option 1"
      value="option1"
    />
    ```

    2. With onChange handler:
    ```tsx
    const handleChange = (value: string) => {
      console.log('Selected value:', value);
    };
    
    <RadioButton
      id="option1"
      name="group1"
      label="Option 1"
      value="option1"
      onChange={handleChange}
    />
    ```
  </StepItem>
</Step>

## Props

| Prop           | Type      | Description                                        |
|----------------|-----------|---------------------------------------------------|
| id             | string    | Unique identifier for the radio button             |
| name           | string    | Group name for the radio button set               |
| label          | string    | Label text displayed next to the radio button     |
| value          | string    | Value associated with the radio button            |
| defaultChecked | boolean   | Whether the radio button is checked by default    |
| disabled       | boolean   | Whether the radio button is disabled              |
| onChange       | function  | Callback when the radio button selection changes  |
