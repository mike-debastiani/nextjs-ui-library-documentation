    import React from "react";
    import clsx from "clsx";

    type BadgeVariant = "default" | "secondary" | "destructive" | "outline" | "blue";

    interface BadgeProps {
      children: React.ReactNode;
      variant?: BadgeVariant;
      className?: string;
    }

    const baseStyles = "inline-flex items-center rounded-full px-3 py-1 text-sm font-medium transition-colors";

    const variantStyles: Record<BadgeVariant, string> = {
      default: "bg-black text-white",
      secondary: "bg-gray-100 text-black",
      destructive: "bg-red-600 text-white",
      outline: "border border-gray-300 text-black bg-white",
      blue: "bg-blue-600 text-white",
    };

    export const Badge: React.FC<BadgeProps> = ({
      children,
      variant = "default",
      className,
    }) => {
      return (
        <span className={clsx(baseStyles, variantStyles[variant], className)}>
          {children}
        </span>
      );
    };

## Setup & Usage

<Step>
  <StepItem title="Create a new Component">
    Create a new component in the `@/components/ui-library` directory with the file name `Badge.tsx`.
  </StepItem>

  <StepItem title="Copy the Full Code inside the component">
    Copy the full code from the "Full Code" tab above and paste it into the `Badge.tsx` file.
  </StepItem>

  <StepItem title="Use the Component">
    Use the Badge component in your code with the desired props:
    ```tsx
    <Badge variant="default">Status</Badge>
    ```
  </StepItem>
</Step>

## Props

| Prop | Type | Description |
|------|------|-------------|
| children | React.ReactNode | The content to display inside the badge (required) |
| variant | "default" \| "secondary" \| "destructive" \| "outline" \| "blue" | Visual style variant (default: "default") |
| className | string | Additional CSS classes to apply |
