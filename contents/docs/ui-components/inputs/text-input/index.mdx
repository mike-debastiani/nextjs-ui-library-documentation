## Codes

```tsx live
function Example() {
  const [value, setValue] = useState("");

  return (
    <TextInputs
      id="example"
      label="Your name"
      placeholder="Enter your name"
      value={value}
      onChange={(e) => setValue(e.target.value)}
    />
  );
}

export default Example;
```
